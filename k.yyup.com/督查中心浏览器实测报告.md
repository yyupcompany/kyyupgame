// 跳转并展开当月
const scrollToCurrentMonth = () => {
  const currentMonth = new Date().getMonth() + 1;
  activeMonths.value = [`${selectedYear.value}-${currentMonth}`];
  // 滚动到当月面板

    document.querySelector(`#month-${currentMonth}`)?.scrollIntoView({
