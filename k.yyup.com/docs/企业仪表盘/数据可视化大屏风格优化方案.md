## 🎨 视觉风格

### 1. 配色方案

#### 背景色
```scss
// 主背景：深蓝色渐变
background: linear-gradient(180deg, #0a1929 0%, #1e3a5f 100%);

// 卡片背景：半透明深色 + 毛玻璃效果
background: rgba(15, 23, 42, 0.6);
backdrop-filter: blur(10px);
```

#### 边框和发光
```scss
// 霓虹蓝色边框
border: 1px solid rgba(59, 130, 246, 0.3);

// 发光效果
box-shadow: 
  0 0 20px rgba(59, 130, 246, 0.2),
  inset 0 0 20px rgba(59, 130, 246, 0.05);

// 悬停发光
&:hover {
  border-color: rgba(59, 130, 246, 0.6);
  box-shadow: 0 0 30px rgba(59, 130, 246, 0.4);
}
```

#### 文字颜色
```scss
// 标题：亮蓝色 + 发光
color: #60a5fa;
text-shadow: 0 0 10px rgba(96, 165, 250, 0.5);

// 数值：白色 + 发光
color: #fff;
text-shadow: 0 0 10px rgba(255, 255, 255, 0.3);

// 次要文字：浅灰蓝
color: #94a3b8;
```

### 2. 卡片大小层级

#### 大卡片（最重要）
- **尺寸**: 占据2-3个网格单元
- **内容**: 客户池中心、活动中心、招生中心
- **特点**: 
  - 大号图标（80px）
  - 大号数字（48px）
  - 包含图表可视化
  - 强烈的发光效果

#### 中卡片（重要）
- **尺寸**: 占据1-2个网格单元
- **内容**: 教学中心、考勤中心、财务中心、任务中心、督查中心
- **特点**:
  - 中号图标（60px）
  - 中号数字（32px）
  - 简单的图表或进度条
  - 中等发光效果

#### 小卡片（辅助）
- **尺寸**: 占据1个网格单元
- **内容**: 其他9个中心
- **特点**:
  - 小号图标（40px）
  - 小号数字（24px）
  - 纯数据展示
  - 轻微发光效果

```typescript
// 使用ECharts创建环形进度
const createRingChart = (value: number, total: number) => {
  return {
    series: [{
      type: 'gauge',
      startAngle: 90,
      endAngle: -270,
      radius: '80%',
      pointer: { show: false },
      progress: {
        show: true,
        overlap: false,
        roundCap: true,
        clip: false,
        itemStyle: {
          color: {
            type: 'linear',
            x: 0, y: 0, x2: 1, y2: 1,
            colorStops: [
              { offset: 0, color: '#3b82f6' },
              { offset: 1, color: '#60a5fa' }
            ]
          }
        }
      },
      axisLine: {
        lineStyle: {
          width: 12,
          color: [[1, 'rgba(59, 130, 246, 0.1)']]
        }
      },
      axisTick: { show: false },
      splitLine: { show: false },
      axisLabel: { show: false },
      detail: {
        valueAnimation: true,
        formatter: '{value}',
        color: '#60a5fa',
        fontSize: 32,
        fontWeight: 'bold',
        offsetCenter: [0, 0]
      },
      data: [{ value: (value / total * 100).toFixed(0) }]
    }]
  }
}
```
