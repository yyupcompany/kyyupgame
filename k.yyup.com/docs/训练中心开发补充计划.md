class TrainingRecommendationEngine {

  recommendByAssessment(
    assessmentResult: AssessmentResult,
    childProfile: ChildProfile
  ): TrainingPlan[] {
    const gaps = this.identifyAbilityGaps(assessmentResult);
    const suitableActivities = this.findSuitableActivities(gaps, childProfile.age);
    return this.generateTrainingPlans(suitableActivities, childProfile);
  }

  // è¯†åˆ«èƒ½åŠ›ç¼ºå£
  private identifyAbilityGaps(assessment: AssessmentResult): AbilityGap[] {
    const gaps: AbilityGap[] = [];
    const threshold = 60; // 60åˆ†ä»¥ä¸‹è®¤ä¸ºéœ€è¦æå‡

    Object.entries(assessment.dimensionScores).forEach(([dimension, score]) => {

          dimension,
          currentScore: score.score,
          targetScore: Math.min(score.score + 20, 100),
          priority: this.calculatePriority(dimension, score.score)
        });
      }
    });

    return gaps.sort((a, b) => b.priority - a.priority);
  }

  // æŸ¥æ‰¾åˆé€‚çš„è®­ç»ƒæ´»åŠ¨
  private findSuitableActivities(
    gaps: AbilityGap[],
    childAge: number

        activity.activityType === this.mapDimensionToType(gap.dimension) &&
        activity.targetAgeMin <= childAge &&
        activity.targetAgeMax >= childAge &&
        activity.difficultyLevel <= this.calculateOptimalDifficulty(gap.currentScore)
      );
    });
  }

  // ç”Ÿæˆè®­ç»ƒè®¡åˆ’
